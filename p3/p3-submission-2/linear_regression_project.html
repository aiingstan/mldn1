def matxMultiply(A, B):
    (rwn_a, cln_a) = shape(A)
    (rwn_b, cln_b) = shape(B)
    if not cln_a == rwn_b:
        raise ValueError
    return [[sum([A[m][n] * B[n][k] for n in range(cln_a)]) for k in range(cln_b)] for m in range(rwn_a)]

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    if not len(A) == len(b):
        return None
    Ab = augmentMatrix(A, b)
    rwn, cln = shape(A)
    for n in range(cln):
        lead = Decimal(0)
        lead_row = 0
        for m in range(n, rwn):
            if abs(Ab[m][n]) > lead:
                lead = Decimal(abs(Ab[m][n]))
                lead_row = m
        if lead < epsilon:
            return None
        if not lead_row == n:
            swapRows(Ab, lead_row, n)
        scaleRow(Ab, n, 1.0/Ab[n][n])
        for m in range(rwn):
            if not m == n and not Ab[m][n] == 0:
                addScaledRow(Ab, m, n, -1.0*Ab[m][n])
    res = [[Ab[m][cln] for m in range(rwn)]]
    matxRound(res, decPts)
    return transpose(res)

def linearRegression2D(X,Y):
    argX = [[X[n]] + [1] for n in range(len(X))]
    Y = transpose([Y])
    B = matxMultiply(transpose(argX), Y)
    A = matxMultiply(transpose(argX), argX)
    h = gj_Solve(A, B)
    return float(h[0][0]), float(h[1][0])
